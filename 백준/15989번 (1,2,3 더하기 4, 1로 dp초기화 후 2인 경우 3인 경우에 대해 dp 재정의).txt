import sys
input = sys.stdin.readline

TC = int(input())

while TC:
    N = int(input())

    dp = [1]*(N+1)
    #dp 배열은 0부터 N까지 총 N+1 길이의 배열이다.
    #dp 배열을 모두 1로 초기화 하는 것으로
    #1만 이용한 가지수를 포함한 것으로 만든다.
    #즉, 모든 숫자는 1만 이용하여 덧셈을 표현하는 경우의 수가 1가지이므로
    #항상 모든 숫자는 1가지 경우를 들고 시작한다고 볼 수 있다.

    for i in range(2, N+1):
        #2를 이용해 덧셈으로 표시할 수 있는 숫자에 대해 2인 경우를 추가하는 것이다.
        #그러므로 범위는 2부터 N+1까지에 해당하고 1로 표현한 경우에 2로 표현한 경우를
        #추가해야하는 것이므로 기존의 1로 표시하는 정보를 담고 있는 dp[i]에
        #2로 표시할 때 1의 개수를 가지고 있는 dp[i-2]의 정보를 더한다.
        dp[i] = dp[i] + dp[i-2]

    for j in range(3, N+1):
        #3을 이용해 덧셈으로 표시할 수 있는 숫자에 대해 3인 경우를 추가하는 것이다.
        #그러므로 범위는 3부터 N+1까지에 해당하고 1로 표현한 경우에 3으로 표현한 경우를
        #추가해야하는 것이므로 기존의 1로 표시하는 정보를 담고 있는 dp[i]에
        #3으로 표시할 때 1의 개수를 가지고 있는 dp[i-3]의 정보를 더한다.
        dp[j] = dp[j] + dp[j-3]

    print(dp[-1])

    TC = TC - 1
