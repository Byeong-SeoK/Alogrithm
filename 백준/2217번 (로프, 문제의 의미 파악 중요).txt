import sys
input = sys.stdin.readline

N = int(input())

rope = []
for _ in range(0, N):
    rope.append(int(input()))

#로프가 받아들일 수 있는 중량이 서로 다르더라도 특정 물체를 들때
#항상 동일한 중량을 로프가 받게 된다. 그래서 예시에 나와있는
#10, 15에서 20이라는 중량의 무게를 들면 10도 10만큼의 중량을 받고 15도 10만큼의 중량을 받게 된다.
#만약 30 중량의 물체를 들면 10, 15가 받는 중량이 각각 15가 된다. 그래서 10인 로프가 버티지 못한다.
#그러므로 10 만큼만 받을 수 있는 중량 20이 최대 중량이 된다.
#그리고 각 로프가 받는 중량의 크기는 총 중량 / 로프의 개수 이다.

rope.sort()
#로프를 오름차순으로 정렬한다.
#왜냐하면 제일 작은 중량의 로프를 사용할 때는 모든 로프를 사용할 수 있고
#조금 더 무거운 중량의 로프를 사용할 때는 그 보다 작은 로프는 사용할 수 없기 때문이다.
#즉, 사용할 수 있는 로프의 개수가 어떤 로프를 사용할 때마다 달라지기 때문에
#배열을 정렬하고 가장 작은 로프의 중량 X 사용할 수 있는 로프의 개수 를 통해 가장 무거운 중량을 구한다.

weight = 0
for i in range(0, len(rope)):
    total = rope[i]*(len(rope)-i)

    if(weight < total):
        weight = total

print(weight)
