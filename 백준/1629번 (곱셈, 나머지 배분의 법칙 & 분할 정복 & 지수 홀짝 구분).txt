input = sys.stdin.readline


def divide(base, exponent):
    if(exponent == 1):
        #더 이상 분할이 불가능하므로 이때의 나머지를 반환한다.
        remain = base % C
        return remain
    else:
        number = divide(base, exponent//2)
        #분할할 때 지수가 1이 될 때까지 분할하도록 한다.

        if(exponent % 2 == 0):
            #만약 분할이 지수가 2->1로 진행되었다고 하자
            #지수가 1일 때의 나머지는 저 위의 분기문에서 반환되므로 1일 때의 나머지 값을 알고 있고
            #그 값이 반환되었으므로 이 분기문에서의 지수 값은 2인 상황이다.
            #이때 지수가 2인 상황에서의 나머지 값은 1일 때 나머지 값을 서로 곱하고 그것을 C로 나누었을 때이다.
            #그러므로 number을 서로 곱하고 그 값을 C로 나누어 나머지를 구하도록 한다.
            remain = (number*number)%C
            return remain
        else:
            #만약 분할이 지수가 5->2로 진행되었다고 하자
            #지수가 1일 때의 나머지는 저 위의 분기문에서 반환되므로 2일 때의 나머지 값을 알고 있고
            #그 값이 반환되었으므로 이 분기문에서의 지수 값은 5인 상황이다.
            #다만 이때 지수 값이 2일 때의 나머지를 알고 있지 1일 때의 나머지는 알고 있지 못한다.
            #왜냐하면 지수가 (2+2+1)로 분할되었기 때문이다.
            #그리고 지수가 각 2일 때의 나머지는 이전 재귀함수들을 통해서 나머지 값을 알고 있다.
            #그래서 2일 때의 나머지 number를 서로 곱하고 지수가 1일 때의 밑을 의미하는 base를 곱하고
            #그 값에 대해 C로 나누어 나머지를 구하도록 한다.
            remain = (number*number*base)%C
            return remain
