N = int(input())

count = 1 #최초의 1번째 칸
end = 1 #1에서 1칸만 옮기면 갈 수 있는 가장 큰 값의 칸
#위 end를 배열로 구현하면 1000000000 같은 아주 큰 수에 대해
#append할 때마다 메모리 재할당에 따른 메모리 초과가 발생할 수 있다.
#start와 같은 시작점에 대한 정보는 end+1에 해당하므로 중복적으로 정보를 저장하는 상황이 발생한다.
#그래서 start에 대한 정보는 신경을 쓸 필요가 없다.

while N > end:
    #1,7,19,37.... 순서로 진행되는데 해당 값보다 더 커질때 멈춘다.
    #7까지 일 때는 2칸이고 19까지 일 때는 3칸인데 만약 입력이 13일 때
    #13은 7보다 크고 19보다 작다. 이때 13 > 7이므로 count = 2를 넘어가고
    #count=count+1 때문에 count는 3의 값을 갖게 된다.
    #그리고 다음 값은 19인데 13 < 19이므로 while문을 탈출하게 된다.
    end = end+(6*count)
    count = count+1
print(count)
